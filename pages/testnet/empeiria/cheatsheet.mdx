# Cheat Sheet
A Tendermint cheat sheet is a quick reference for essential commands to control Tendermint blockchains.

## Managing keys

```js filename="Generate new key"
emped keys add wallet
```

```js filename="Recover key"
emped keys add wallet --recover
```
```js filename="List all key"
emped keys list
```
```js filename="Delete key"
emped keys delete wallet
```
```js filename="Export key"
emped keys export wallet
```
```js filename="Import key"
emped keys import wallet wallet.backup
```
```js filename="Query wallet balances"
emped q bank balances $(emped keys show wallet -a)
```

## Managing validators
```js filename="Create validator"
emped tx mstaking create-validator \
--amount 1000000uempe \
--from $WALLET \
--commission-rate 0.1 \
--commission-max-rate 0.2 \
--commission-max-change-rate 0.01 \
--min-self-delegation 1 \
--pubkey $(emped tendermint show-validator) \
--moniker "YOUR-MONIKER" \
--identity "" \
--website "" \
--details "Service by zulna" \
--chain-id empe-testnet-2 \
--gas auto --fees 30uempe \
-y
```

```js filename="Edit validator"
emped tx mstaking edit-validator \
--commission-rate 0.1 \
--new-moniker "$MONIKER" \
--identity "" \
--details "Service by zulna" \
--from $WALLET \
--chain-id empe-testnet-2 \
--gas auto --fees 30uempe \
-y
```

```js filename="Unjail validator"
emped tx slashing unjail --from wallet --chain-id empe-testnet-2 --gas-adjustment 1.5 --gas auto --fees 30uempe -y
```

```js filename="Validator jail reason"
emped query slashing signing-info $(emped tendermint show-validator)
```
```js filename="List active validator"
emped q staking validators -oj --limit=3000 | jq '.validators[] | select(.status=="BOND_STATUS_BONDED")' | jq -r '(.tokens|tonumber/pow(10; 6)|floor|tostring) + " \t " + .description.moniker' | sort -gr | nl
```
```js filename="List incative validator"
emped q staking validators -oj --limit=3000 | jq '.validators[] | select(.status=="BOND_STATUS_UNBONDED")' | jq -r '(.tokens|tonumber/pow(10; 6)|floor|tostring) + " \t " + .description.moniker' | sort -gr | nl
```
```js filename="View validator details"
emped q staking validator $(emped keys show wallet --bech val -a)
```

## Managing Tokens

```js filename="Withdraw reward from all validator"
emped tx distribution withdraw-all-rewards --from wallet --chain-id empe-testnet-2 --gas-adjustment 1.5 --gas auto --fees 30uempe -y
```
```js filename="Withdraw reward and commission"
emped tx distribution withdraw-rewards $(emped keys show wallet --bech val -a) --commission --from wallet --chain-id empe-testnet-2 --gas-adjustment 1.5 --gas auto --fees 30uempe -y
```
```js filename="Delegate tokens to your validator"
emped tx staking delegate $(emped keys show wallet --bech val -a) 1000000uempe --from wallet --chain-id empe-testnet-2 --gas-adjustment 1.5 --gas auto --fees 30uempe -y
```
```js filename="Delegate token to other validator, change <to-valoper-address> as you like"
emped tx staking delegate <to-valoper-address> 1000000uempe --from wallet --chain-id empe-testnet-2 --gas-adjustment 1.5 --gas auto --fees 30uempe -y
```
```js filename="Redelegate to another validator"
emped tx staking redelegate $(emped keys show wallet --bech val -a) <to-valoper-address> 1000000uempe --from wallet --chain-id empe-testnet-2 --gas-adjustment 1.5 --gas auto --fees 30uempe -y
```
```js filename="Unbond token from your own validator"
emped tx staking unbond $(emped keys show wallet --bech val -a) 1000000uempe --from wallet --chain-id empe-testnet-2 --gas-adjustment 1.5 --gas auto --fees 30uempe -y
```
```js filename="Send token to the wallet"
emped tx bank send wallet <to-wallet-address> 1000000uempe --from wallet --chain-id empe-testnet-2 --gas-adjustment 1.5 --gas auto --fees 30uempe -y
```

## Usefulness

```js filename="Update custom port"
CUSTOM_PORT=239
sed -i -e "s%^proxy_app = \"tcp://127.0.0.1:26658\"%proxy_app = \"tcp://127.0.0.1:${CUSTOM_PORT}58\"%; s%^laddr = \"tcp://127.0.0.1:26657\"%laddr = \"tcp://127.0.0.1:${CUSTOM_PORT}57\"%; s%^pprof_laddr = \"localhost:6060\"%pprof_laddr = \"localhost:${CUSTOM_PORT}60\"%; s%^laddr = \"tcp://0.0.0.0:26656\"%laddr = \"tcp://0.0.0.0:${CUSTOM_PORT}56\"%; s%^prometheus_listen_addr = \":26660\"%prometheus_listen_addr = \":${CUSTOM_PORT}66\"%" $HOME/.empe-chain/config/config.toml
sed -i -e "s%^address = \"tcp://0.0.0.0:1317\"%address = \"tcp://0.0.0.0:${CUSTOM_PORT}17\"%; s%^address = \":8080\"%address = \":${CUSTOM_PORT}80\"%; s%^address = \"0.0.0.0:9090\"%address = \"0.0.0.0:${CUSTOM_PORT}90\"%; s%^address = \"0.0.0.0:9091\"%address = \"0.0.0.0:${CUSTOM_PORT}91\"%" $HOME/.empe-chain/config/app.toml
```
```js filename="Disable indexer"
sed -i -e 's|^indexer *=.*|indexer = "null"|' $HOME/.empe-chain/config/config.toml
```
```js filename="Enable indexer"
sed -i -e 's|^indexer *=.*|indexer = "kv"|' $HOME/.empe-chain/config/config.toml
```
```js filename="Pruning update"
sed -i \
  -e 's|^pruning *=.*|pruning = "custom"|' \
  -e 's|^pruning-keep-recent *=.*|pruning-keep-recent = "100"|' \
  -e 's|^pruning-keep-every *=.*|pruning-keep-every = "0"|' \
  -e 's|^pruning-interval *=.*|pruning-interval = "19"|' \
  $HOME/.empe-chain/config/app.toml
```

## Maintenance

```js filename="Get validator information"
emped status 2>&1 | jq .ValidatorInfo
```
```js filename="Get sync information"
emped status 2>&1 | jq .SyncInfo
```
```js filename="Get node peer"
echo $(emped tendermint show-node-id)'@'$(curl -s ifconfig.me)':'$(cat $HOME/.empe-chain/config/config.toml | sed -n '/Address to listen for incoming connection/{n;p;}' | sed 's/.*://; s/".*//')
```
```js filename="Check validator keys"
[[ $(emped q staking validator $(emped keys show wallet --bech val -a) -oj | jq -r .consensus_pubkey.key) = $(emped status | jq -r .ValidatorInfo.PubKey.value) ]] && echo -e "\n\e[1m\e[32mTrue\e[0m\n" || echo -e "\n\e[1m\e[31mFalse\e[0m\n"
```
```js filename="Get live peers"
curl -sS http://localhost:23957/net_info | jq -r '.result.peers[] | "\(.node_info.id)@\(.remote_ip):\(.node_info.listen_addr)"' | awk -F ':' '{print $1":"$(NF)}'
```
```js filename="Configure minimum gas prices"
sed -i -e "s/^minimum-gas-prices *=.*/minimum-gas-prices = \"10mpx\"/" $HOME/.empe-chain/config/app.toml
```
```js filename="Enable prometheus"
sed -i -e "s/prometheus = false/prometheus = true/" $HOME/.empe-chain/config/config.toml
```
```js filename="Reset chain data"
emped tendermint unsafe-reset-all --keep-addr-book --home $HOME/.empe-chain --keep-addr-book
```
```js filename="Remove node"
sudo systemctl stop emped
sudo systemctl disable emped
sudo rm -rf /etc/systemd/system/emped.service
sudo rm $(which emped)
sudo rm -rf $HOME/.warden
sed -i "EMPED_/d" $HOME/.bash_profile
```
